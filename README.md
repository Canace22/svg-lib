# Vue-Svg-Icons

### Download

```
yarn add vue-svg-icons || npm install vue-svg-icons
```

### Use

```js
import {IconBase, IconCircle} from 'vue-svg-icons'
```

```html
<icon-base width="100" height="80">
  <icon-circle r="5"></icon-circle>
</icon-base>
```
